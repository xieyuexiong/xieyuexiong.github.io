第一次的工程实践项目：TC图形库下的贪吃蛇小程序（需要安装easyX）
从http://www.easyx.cn/ 下载后安装

我的源代码如下：

#include<stdio.h>

#include <graphics.h>

#include <conio.h>

#include <stdlib.h>

#include <time.h>
/*初始化操作*/
//宏定义上下左右键值

#define up 72

#define down 80

#define left 75

#define right 77

//每一个结点代表一节蛇身，采用双向链表方便定位结点
typedef struct Snake
{
int x;
int y;
struct Snake next,prior;

}snake;

//flag代表食物是否被吃 1表示吃了 0表示没吃
struct Food
{
int x;
int y;
int flag;
}food;

//先定义开始的四节蛇
snake head,tail,p,q;
snake s,_new;

int direct=right; //初始化方向向左
int length=4;
int i;
int score=-10; //初始化分数
int speed=0; //游戏速度

/***/

void Chooselevel(); //选择难度
void Initsnake(); //初始化蛇
void Drawsnake(); //画蛇
void Movesnake(); //移动蛇的坐标
void Controlsnake(); //键盘控制蛇的移动
void Creatfood(); //随机出现食物
void Drawfood(); //画食物
void Eatingfood(); //蛇吃到食物
void Endgame(); //游戏结束

int main()
{
Chooselevel();
Initsnake();
while(1)
{
while(!kbhit())
{

            if(food.flag==0)
            {
                Creatfood();
                score+=10;
            }
            cleardevice();
            loadimage(NULL, _T("背景.jpg"));
            Movesnake();
            Drawsnake();
            Drawfood();
            Eatingfood();
            Sleep(speed);
        }
        Controlsnake();
    }
return 0;

}

void Initsnake()
{
//初始化一个640*480的窗口
initgraph(640,480);

//给初始的四节蛇分配空间 
head = (snake *)malloc(sizeof(snake));
tail = (snake *)malloc(sizeof(snake));
p = (snake*)malloc(sizeof(snake));
q = (snake*)malloc(sizeof(snake));

//初始化坐标 
head->x=60;
head->y=30;
p->x = 50;
p->y = 30;
q->x = 40;
q->y = 30;
tail->x=30;
tail->y=30;

//创建链表 
head->next = p;
p->next = q;
q->next = tail;
tail->next = NULL;
tail->prior=q;
q->prior=p;
p->prior=head;
head->prior=NULL;

}

void Drawsnake()
{
//设置蛇的颜色
setlinecolor(RED);
setfillcolor(YELLOW);

//遍历蛇身进行绘图 
s=head;
for(i=0;i<length;i++)
{
    fillrectangle(s->x,s->y,s->x+10,s->y+10);
    s=s->next; 
}

}

void Movesnake()
{
//将蛇的坐标向下一步进行覆盖
s=head;
while(1)
{
if(s->next==NULL) break;
else
s=s->next;
}
for(i=0;i<length-1;i++)
{

    s->x=s->prior->x;
    s->y=s->prior->y;
    s=s->prior;
}

//移动蛇头 
switch(direct)
{
case up:
    head->y -= 10;
    break;
case down:
    head->y += 10;
    break;
case left:
    head->x -= 10;
    break;
case right:
    head->x += 10;
    break;
}

}

void Controlsnake()
{
int move;
move=getch();

//获取键盘输入判断移动方向 
switch(move)
{
case up:
    if(direct!=down)
    direct=up;
    break;
case down:
    if(direct!=up)
    direct=down;
    break;
case left:
    if(direct!=right)
    direct=left;
    break;
case right:
    if(direct!=left)
    direct=right;
    break;
}

}

void Creatfood()
{
//随机生成食物坐标
srand((unsigned int)time(NULL));
food.x=rand()%6410;
food.y=rand()%4810;

//遍历蛇身 排除食物出现在蛇身 
s=head;
for(i=0;i<length;i++)
{
    if(s->x==food.x&&s->y==food.y)
    {
        food.flag=0;
    }
    else
    {
        food.flag = 1;
    }
    s=s->next;
}

}

void Drawfood()
{
//设置食物颜色
setlinecolor(LIGHTRED);
setfillcolor(LIGHTRED);
fillroundrect(food.x,food.y,food.x+10,food.y+10,15,15);
}

void Eatingfood()
{
//判断食物和蛇头是否重合
if(head->x==food.x&&head->y==food.y)
{
s=head;
while (1)
{
if(s->next==NULL) break;
s=s->next;
}
_new=(snake*)malloc(sizeof(snake));
_new->next=s->next;
s->next=_new;
_new->prior=s;
length++;
food.flag=0;
}

//蛇如果吃到自己则退出 
s=tail;
for(i=3;i<length;i++)
{
    if((head->x == s->x&&head->y == s->y+10&&direct == down)||(head->x == s->x&&head->y == s->y-10&&direct == up)
        ||(head->x == s->x-10&&head->y == s->y&&direct == left)||(head->x == s->x+10&&head->y == s->y&&direct == right))
    {
        while(1)
        {
            Endgame();
        }
    }
    s->next;
}

//蛇如果撞到墙则退出游戏 
if(head->x==-10||head->x==640||head->y==-10||head->y==480)
{
    while(1)
    {
        Endgame();
    }
}

}

void Endgame()
{
//设置文字格式
char temp[20];
itoa(score,temp,10);
settextcolor(RED);
settextstyle(50,25,”楷体”);

//结束界面 
cleardevice();
outtextxy(210,80,"游戏结束");
outtextxy(150,210,"你的分数是：");
outtextxy(450,210,temp);
Sleep(10000);

}

void Chooselevel()
{

char str[20];
printf("请输入难度（easy,normal,hard）：");
scanf("%s",str);
if(strcmp(str,"easy")==0)
{
    speed=300;
}
else if(strcmp(str,"normal")==0)
{
    speed=100;
}
else if(strcmp(str,"hard")==0)
{
    speed=50;
}
else
{
    printf("输入错误！");
    exit(0);
}

}
